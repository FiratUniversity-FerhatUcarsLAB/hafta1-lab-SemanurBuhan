GÖREV-1: Favori oyunda yüksek skor alma algoritması 

// 1. BAŞLANGIÇ DEĞİŞKENLERİ VE SABİTLERİN TANIMLANMASI
DEĞİŞKENLER:
    SKOR = 0
    CAN = 3
    YÜKSEK_SKOR = oku_kayıtlı_yüksek_skor()
    SEVİYE = 1
    Oyun_Devam_Ediyor = DOĞRU
    ÖZEL_GÜÇ_AKTİF = YANLIŞ

SABİTLER:
    DÜŞMAN_YOK_ETME_SKORU = 100
    CAN_KAYBI_TEHLİKE_EŞİĞİ = 1
    YENİ_SEVİYE_SKOR_EŞİĞİ = 500

// 2. ANA OYUN DÖNGÜSÜ (TEKRAR EDEN İŞLEM)
BAŞLA:
    TEKRARLA (Oyun_Devam_Ediyor DOĞRU olduğu sürece):

        // 2.1 Oyuncu Girişi (Koşullu İşlem)
        GİRİŞ = oku_oyuncu_girişi()
        EĞER (GİRİŞ == "ATEŞ"):
            işle_ateş_etme()
        EĞER (GİRİŞ == "ÖZEL_GÜÇ_KULLAN" VE ÖZEL_GÜÇ_AKTİF == DOĞRU):
            aktive_et_özel_güç()
            ÖZEL_GÜÇ_AKTİF = YANLIŞ

        // 2.2 Oyun Durumunun Güncellenmesi
        güncelle_düşman_konumları()
        güncelle_mermi_konumları()

        // 2.3 Çarpışma ve Skor Kontrolü (Tekrar Eden ve Koşullu İşlemler)
        HER_MERMİ_İÇİN (Oyun_Alanındaki_Mermiler):
            HER_DÜŞMAN_İÇİN (Oyun_Alanındaki_Düşmanlar):
                EĞER (çarpışma_var(MERMİ, DÜŞMAN)):
                    // Skor Kazanımı
                    SKOR = SKOR + DÜŞMAN_YOK_ETME_SKORU * SEVİYE
                    
                    // Seviye Atlama ve Bonus Kontrolü
                    EĞER (SKOR >= SEVİYE * YENİ_SEVİYE_SKOR_EŞİĞİ):
                        SEVİYE = SEVİYE + 1
                        göster_mesaj("SEVİYE ATLANDI: " + SEVİYE)
                        DÜŞMAN_YOK_ETME_SKORU = DÜŞMAN_YOK_ETME_SKORU + 25
                        çoğalt_düşman_sayısı()
                    
                    // Özel Güç Kontrolü
                    EĞER (SKOR MOD 1000 == 0):
                        ÖZEL_GÜÇ_AKTİF = DOĞRU
                        göster_mesaj("ÖZEL GÜÇ KAZANDIN!")
                    
                    DÜŞMAN.durum = "YOK_EDİLDİ"
                    MERMİ.durum = "YOK_EDİLDİ"
                    SONUÇ: devam_et

        // 2.4 Oyuncu Can Kontrolü (Koşullu Durum)
        EĞER (oyuncu_hasar_aldı()):
            CAN = CAN - 1
            oynat_hasar_animasyonu()

            EĞER (CAN <= CAN_KAYBI_TEHLİKE_EŞİĞİ):
                göster_uyarı_mesajı("DİKKAT: Canın Çok Az!")

        // 2.5 Oyun Sonu Kontrolü (Koşullu Çıkış)
        EĞER (CAN <= 0):
            Oyun_Devam_Ediyor = YANLIŞ // Ana döngüyü sonlandır

    // ANA DÖNGÜ SONU

    // 3. YÜKSEK SKOR KAYIT KONTROLÜ
    GÖSTER_YÜKSEK_SKOR_EKRANI()
    
    EĞER (SKOR > YÜKSEK_SKOR):
        YÜKSEK_SKOR = SKOR
        kaydet_yeni_yüksek_skor(YÜKSEK_SKOR)
        göster_mesaj("YENİ YÜKSEK SKOR: " + YÜKSEK_SKOR + "!")
    AKSİ TAKDİRDE:
        göster_mesaj("Skor: " + SKOR + " (Yüksek Skor: " + YÜKSEK_SKOR + ")")

    ÇIKIŞ
